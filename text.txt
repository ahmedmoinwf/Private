@InjectMocks
private WebClientConfiguration webClientConfiguration;

@Test
void addCommonRequestHeaderValues_shouldAddHeadersToHttpHeaders() {
    // Act
    Consumer<HttpHeaders> headerConsumer = (Consumer<HttpHeaders>) ReflectionTestUtils.invokeMethod(
            webClientConfiguration, "addCommonRequestHeaderValues");

    HttpHeaders headers = new HttpHeaders();
    headerConsumer.accept(headers);

    // Assert
    assertTrue(headers.containsKey("WF-senderMessageId"));
    assertNotNull(headers.getFirst("WF-senderMessageId"));
    assertTrue(UUID.fromString(headers.getFirst("WF-senderMessageId")) instanceof UUID);

    assertTrue(headers.containsKey("WF-creationTimeStamp"));
    assertNotNull(headers.getFirst("WF-creationTimeStamp"));

    assertTrue(headers.containsKey("WF-senderApplicationId"));
    assertEquals("1WOC", headers.getFirst("WF-senderApplicationId"));

    assertTrue(headers.containsKey("WF-senderHostName"));
    assertEquals("localhost", headers.getFirst("WF-senderHostName"));
}
